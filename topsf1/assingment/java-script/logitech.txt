 9. CREATE AUDIT TABLE FOR TRIGGERS

CREATE TABLE model_audit (
    audit_id INT PRIMARY KEY AUTO_INCREMENT,
    action VARCHAR(50),
    model_name VARCHAR(100),
    time TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

 TRIGGER FOR INSERT

DELIMITER $$

CREATE TRIGGER time
AFTER INSERT ON model
FOR EACH ROW
BEGIN
    INSERT INTO model_audit(action_performed, model_name)
    VALUES ('INSERT', NEW.model_name);
END $$


 TRIGGER FOR UPDATE

CREATE TRIGGER timeb
AFTER UPDATE ON models
FOR EACH ROW
BEGIN
    INSERT INTO model_audit(action_performed, model_name)
    VALUES ('UPDATE', NEW.model_name);
END $$

 TRIGGER FOR DELETE

CREATE TRIGGER timec
AFTER DELETE ON model
FOR EACH ROW
BEGIN
    INSERT INTO model_audit(action_performed, model_name)
    VALUES ('DELETE', OLD.model_name);
END $$

DELIMITER ;

 CHECK AUDIT RECORDS

SELECT * FROM models_audit;
